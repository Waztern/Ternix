function doLogin() { const u = document.getElementById('u-name').value; const p = document.getElementById('u-pass').value; const err = document.getElementById('err'); // Проверка ника: от 3 до 20 символов, только английские буквы и цифры const nameRegex = /^[a-zA-Z0-9]{3,20}$/; // Проверка пароля: минимум 8 символов const passRegex = /^.{8,}$/; if(!nameRegex.test(u)) { err.innerText = "Username: 3-20 chars, English & numbers only!"; return; } if(!passRegex.test(p)) { err.innerText = "Password: minimum 8 characters!"; return; } localStorage.setItem('ternix_user', u); localStorage.setItem('ternix_firens', '0'); localStorage.setItem('ternix_nex', '100'); window.location.reload(); // Перезагрузка для входа }
